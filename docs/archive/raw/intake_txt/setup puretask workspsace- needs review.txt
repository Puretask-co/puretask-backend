// src/state/jobStateMachine.ts

export type JobStatus =
  | "requested"
  | "approved"
  | "assigned"
  | "in_progress"
  | "completed"
  | "cancelled";

export type JobTransitionEvent =
  | "job_approved"
  | "job_assigned"
  | "job_started"
  | "job_completed"
  | "job_cancelled";

export interface JobTransitionContext {
  role: "client" | "cleaner" | "admin";
}

/**
 * All allowed transitions in the system.
 *
 * Key: current_status
 * Value: allowed transition events, and which status they lead to.
 */
const transitions: Record<
  JobStatus,
  Partial<Record<JobTransitionEvent, JobStatus>>
> = {
  requested: {
    job_approved: "approved",
    job_cancelled: "cancelled",
  },
  approved: {
    job_assigned: "assigned",
    job_cancelled: "cancelled",
  },
  assigned: {
    job_started: "in_progress",
    job_cancelled: "cancelled",
  },
  in_progress: {
    job_completed: "completed",
    job_cancelled: "cancelled",
  },
  completed: {
    // no further transitions
  },
  cancelled: {
    // no further transitions
  },
};

export function isTerminal(status: JobStatus): boolean {
  return status === "completed" || status === "cancelled";
}

/**
 * Check if a transition is allowed given current status + event type.
 * Throws on invalid transitions.
 */
export function getNextStatusOrThrow(
  currentStatus: JobStatus,
  event: JobTransitionEvent
): JobStatus {
  const allowed = transitions[currentStatus]?.[event];
  if (!allowed) {
    throw new Error(
      Invalid transition: cannot apply ${event} to status ${currentStatus}
    );
  }
  return allowed;
}

/**
 * Optional: some basic role-based checks.
 * You can expand this if you want stricter rules.
 */
export function ensureRoleCanTransition(
  role: "client" | "cleaner" | "admin",
  event: JobTransitionEvent
) {
  // Example rules:
  if (event === "job_approved" && role !== "admin") {
    throw new Error("Only admins can approve jobs");
  }

  if (event === "job_assigned" && role !== "admin") {
    throw new Error("Only admins can assign cleaners");
  }

  if (event === "job_started" && role !== "cleaner") {
    throw new Error("Only cleaners can start jobs");
  }

  if (event === "job_completed" && role !== "cleaner") {
    throw new Error("Only cleaners can complete jobs");
  }

  if (event === "job_cancelled" && role === "cleaner") {
    // let client/admin cancel, but not cleaner
    throw new Error("Cleaners cannot cancel jobs");
  }
}

import { query } from "../db/client";

export type CreditDirection = "credit" | "debit";

export interface CreditLedgerEntryInput {
  userId: string;
  jobId?: string | null;
  amount: number; // positive integer
  direction: CreditDirection;
  reason?: string;
}

/**
 * Insert a single row into credit_ledger.
 */
export async function addCreditLedgerEntry(input: CreditLedgerEntryInput) {
  const { userId, jobId = null, amount, direction, reason } = input;

  if (amount <= 0) {
    throw new Error("amount must be positive");
  }

  const sql = 
    INSERT INTO credit_ledger (user_id, job_id, amount, direction, reason)
    VALUES ($1, $2, $3, $4, $5)
    RETURNING *
  ;

  const result = await query(sql, [userId, jobId, amount, direction, reason ?? null]);
  return result.rows[0];
}

/**
 * Current credit balance for a user:
 *  sum(credit) - sum(debit)
 */
export async function getUserCreditBalance(userId: string): Promise<number> {
  const sql = 
    SELECT
      COALESCE(SUM(CASE WHEN direction = 'credit' THEN amount ELSE 0 END), 0) -
      COALESCE(SUM(CASE WHEN direction = 'debit' THEN amount ELSE 0 END), 0)
      AS balance
    FROM credit_ledger
    WHERE user_id = $1
  ;

  const result = await query(sql, [userId]);
  return result.rows[0]?.balance ?? 0;
}

/**
 * Spend credits on a job (debit).
 * Throws if user does not have enough credits.
 */
export async function spendCreditsForJob(
  userId: string,
  jobId: string,
  amount: number
) {
  const balance = await getUserCreditBalance(userId);

  if (balance < amount) {
    throw new Error("INSUFFICIENT_CREDITS");
  }

  return addCreditLedgerEntry({
    userId,
    jobId,
    amount,
    direction: "debit",
    reason: Job ${jobId} charge,
  });
}

/**
 * Refund credits back to a user for a job (credit).
 */
export async function refundCreditsForJob(
  userId: string,
  jobId: string,
  amount: number
) {
  return addCreditLedgerEntry({
    userId,
    jobId,
    amount,
    direction: "credit",
    reason: Job ${jobId} refund,
  });
}

export async function updateJob(input: UpdateJobInput) {
  const { jobId, scheduledStart, estimatedHours, totalCredits, cleanerId } = input;

  const existing = await getJob(jobId);
  if (!existing) throw new Error("JOB_NOT_FOUND");

  const newScheduled = scheduledStart ?? existing.scheduled_start;
  const newEstimated = estimatedHours ?? existing.estimated_hours;
  const newTotalCredits = totalCredits ?? existing.total_credits;
  const newCleanerId =
    cleanerId !== undefined ? cleanerId : (existing.cleaner_id as string | null);

  const sql = 
    UPDATE jobs
    SET scheduled_start = $2,
        estimated_hours = $3,
        total_credits = $4,
        cleaner_id = $5,
        updated_at = now()
    WHERE id = $1
    RETURNING *
  ;

  const result = await query(sql, [
    jobId,
    newScheduled,
    newEstimated,
    newTotalCredits,
    newCleanerId,
  ]);

  return result.rows[0];

/**
 * Transition job status using the state machine & log event.
 */
export interface TransitionJobInput {
  jobId: string;
  eventType: "job_approved" | "job_assigned" | "job_started" | "job_completed" | "job_cancelled";
  payload?: unknown;
  requesterId: string;
  requesterRole: "client" | "cleaner" | "admin";
}

export async function transitionJobStatus(input: TransitionJobInput) {
  const { jobId, eventType, payload, requesterId, requesterRole } = input;

  const job = await getJob(jobId);
  if (!job) throw new Error("JOB_NOT_FOUND");

  const currentStatus = job.status as JobStatus;

  ensureRoleCanTransition(requesterRole, eventType);
  const nextStatus = getNextStatusOrThrow(currentStatus, eventType as JobTransitionEvent);

  // If cancelling, refund credits to client
  if (eventType === "job_cancelled") {
    await refundCreditsForJob(job.client_id, job.id, job.total_credits);
  }

  const sql = 
    UPDATE jobs
    SET status = $2,
        updated_at = now()
    WHERE id = $1
    RETURNING *
  ;
  const result = await query(sql, [jobId, nextStatus]);
  const updated = result.rows[0];

  await logJobEvent({
    jobId,
    eventName: eventType,
    payload,
    clientId: job.client_id,
    cleanerId: job.cleaner_id,
    source: "backend",
  });

  return updated;
}

  const jobResult = await query(jobSql, [jobId]);
  const job = jobResult.rows[0];

  const eventsSql = 
    SELECT *
    FROM job_events

export async function query<T = unknown>(text: string, params?: any[]): Promise<T[]> {
  const client = await pool.connect();
  try {
    const result = await client.query(text, params);
    return result.rows as T[];
  } finally {
    client.release();
  }

4️⃣ Full state machine (pure logic)

What it is

A pure, testable function that takes current status + action + context and returns next status or throws an error.

// src/lib/stateMachine.ts
import type { JobStatus } from '../types/db';

export type JobAction =
  | 'create'
  | 'accept'
  | 'on_my_way'
  | 'check_in'
  | 'check_out'
  | 'submit_for_approval'
  | 'approve'
  | 'dispute'
  | 'cancel';

interface TransitionContext {
  actor: 'client' | 'cleaner' | 'admin' | 'system';
  hasGps?: boolean;
  hasBeforePhotos?: boolean;
  hasAfterPhotos?: boolean;
  rating?: number;
}

export function transitionJob(
  current: JobStatus,
  action: JobAction,
  ctx: TransitionContext,
): JobStatus {
  switch (action) {
    case 'accept':
      if (ctx.actor !== 'cleaner') throw new Error('Only cleaner can accept');
      if (current !== 'requested') throw new Error('Job must be requested to accept');
      return 'accepted';

    case 'on_my_way':
      if (ctx.actor !== 'cleaner') throw new Error('Only cleaner can go on_my_way');
      if (current !== 'accepted') throw new Error('Job must be accepted to go on_my_way');
      return 'on_my_way';

    case 'check_in':
      if (ctx.actor !== 'cleaner') throw new Error('Only cleaner can check_in');
      if (current !== 'on_my_way') throw new Error('Job must be on_my_way to check_in');
      if (!ctx.hasGps) throw new Error('GPS required to check_in');
      return 'in_progress';

    case 'check_out':
      if (ctx.actor !== 'cleaner') throw new Error('Only cleaner can check_out');
      if (current !== 'in_progress') throw new Error('Job must be in_progress to check_out');
      return 'awaiting_approval';

    case 'submit_for_approval':
      if (ctx.actor !== 'cleaner') throw new Error('Only cleaner can submit_for_approval');
      if (current !== 'awaiting_approval') throw new Error('Job must be awaiting_approval');
      return 'awaiting_approval'; // no status change, but event emitted

    case 'approve':
      if (ctx.actor !== 'client') throw new Error('Only client can approve');
      if (current !== 'awaiting_approval' && current !== 'disputed') {
        throw new Error('Job must be awaiting_approval or disputed to approve');
      }
      if (ctx.rating && (ctx.rating < 1 || ctx.rating > 5)) {
        throw new Error('Rating must be 1-5');
      }
      return 'completed';

    case 'dispute':
      if (ctx.actor !== 'client') throw new Error('Only client can dispute');
      if (current !== 'awaiting_approval') throw new Error('Job must be awaiting_approval to dispute');
      return 'disputed';

    case 'cancel':
      if (current === 'completed') throw new Error('Cannot cancel completed job');
      return 'cancelled';

    default:
      throw new Error('Unsupported action');
  }
}


Pure logic = no DB calls, no HTTP — easy to unit test.

  params: unknown[] = [],
): Promise<T[]> {
  const client = await pool.connect();
  try {
    const result = await client.query(text, params);
    return result.rows as T[];
  } finally {
    client.release();
  }
}

// src/lib/stateMachine.ts
import type { JobStatus } from '../types/db';

export type JobAction =
  | 'create'
  | 'accept'
  | 'on_my_way'
  | 'check_in'
  | 'check_out'
  | 'submit_for_approval'
  | 'approve'
  | 'dispute'
  | 'cancel';

type Actor = 'client' | 'cleaner' | 'admin' | 'system';

interface TransitionRule {
  from: JobStatus[];
  to: JobStatus;
  allowedActors: Actor[];
  extraChecks?: (ctx: TransitionContext) => void;
}

export interface TransitionContext {
  actor: Actor;
  hasGps?: boolean;
  rating?: number;
  hasClientNotes?: boolean;
}

const rules: Record<JobAction, TransitionRule> = {
  accept: {
    from: ['requested'],
    to: 'accepted',
    allowedActors: ['cleaner'],
  },
  on_my_way: {
    from: ['accepted'],
    to: 'on_my_way',
    allowedActors: ['cleaner'],
  },
  check_in: {
    from: ['on_my_way'],
    to: 'in_progress',
    allowedActors: ['cleaner'],
    extraChecks: (ctx) => {
      if (!ctx.hasGps) throw new Error('GPS required to check in');
    },
  },
  check_out: {
    from: ['in_progress'],
    to: 'awaiting_approval',
    allowedActors: ['cleaner'],
  },
  submit_for_approval: {
    from: ['awaiting_approval'],
    to: 'awaiting_approval',
    allowedActors: ['cleaner'],
  },
  approve: {
    from: ['awaiting_approval', 'disputed'],
    to: 'completed',
    allowedActors: ['client', 'admin'],
    extraChecks: (ctx) => {
      if (ctx.rating && (ctx.rating < 1 || ctx.rating > 5)) {
        throw new Error('Rating must be 1–5');
      }
    },
  },
  dispute: {
    from: ['awaiting_approval'],
    to: 'disputed',
    allowedActors: ['client', 'admin'],
    extraChecks: (ctx) => {
      if (!ctx.hasClientNotes) throw new Error('Client notes required to dispute');
    },
  },
  cancel: {
    from: ['requested', 'accepted', 'on_my_way'],
    to: 'cancelled',
    allowedActors: ['client', 'admin'],
  },
  create: {
    from: ['requested'],
    to: 'requested',
    allowedActors: ['client', 'system'],
  },
};

export function transitionJob(
  current: JobStatus,
  action: JobAction,
  ctx: TransitionContext,
): JobStatus {
  const rule = rules[action];
  if (!rule) throw new Error(Unsupported action: ${action});

  if (!rule.from.includes(current)) {
    throw new Error(Cannot ${action} when job status is ${current});
  }

  if (!rule.allowedActors.includes(ctx.actor)) {
    throw new Error(Actor ${ctx.actor} is not allowed to ${action});
  }

  rule.extraChecks?.(ctx);

  return rule.to;
}


Pure, declarative, and easy to extend.

// src/lib/auth.ts
import { Request, Response, NextFunction } from 'express';
import jwt from 'jsonwebtoken';
import { config } from './config';

export interface AuthUser {
  id: string;
  role: 'client' | 'cleaner' | 'admin';
}

declare global {
  namespace Express {
    interface Request {
      user?: AuthUser;
    }
  }
}

export function auth(requiredRole?: AuthUser['role']) {
  return (req: Request, res: Response, next: NextFunction) => {
    const header = req.headers.authorization;
    if (!header?.startsWith('Bearer ')) {
      return res.status(401).json({ error: 'Missing auth token' });
    }

    try {
      const token = header.slice('Bearer '.length);
      const decoded = jwt.verify(token, config.jwtSecret) as AuthUser;
      req.user = decoded;
      if (requiredRole && decoded.role !== requiredRole) {
        return res.status(403).json({ error: 'Forbidden' });
      }
      next();
    } catch {
      return res.status(401).json({ error: 'Invalid auth token' });
    }
  };
}

// src/lib/validation.ts
import { ZodSchema } from 'zod';
import { Request, Response, NextFunction } from 'express';

export function validateBody<T>(schema: ZodSchema<T>) {
  return (req: Request, res: Response, next: NextFunction) => {
    const result = schema.safeParse(req.body);
    if (!result.success) {
      return res.status(400).json({ error: 'Invalid body', details: result.error.issues });
    }
    req.body = result.data;
    next();
  };
}

export async function handleStripeEvent(event: StripeType.Event) {
  await query(
    
      INSERT INTO stripe_events (stripe_event_id, type, payload, processed)
      VALUES ($1, $2, $3, false)
      ON CONFLICT (stripe_event_id) DO NOTHING
    ,
    [event.id, event.type, JSON.stringify(event)],
  );

  switch (event.type) {
    case 'payment_intent.succeeded': {
      const pi = event.data.object as StripeType.PaymentIntent;
      const jobId = pi.metadata.jobId;
      if (!jobId) break;

      // update payment_intents + credit ledger + publish job.payment_succeeded event etc.
      await publishEvent({
        jobId,
        actorType: 'system',
        actorId: null,
        eventType: 'job.payment_succeeded',
        payload: { stripePaymentIntentId: pi.id },
      });
      break;
    }
    // other cases...
  }

  await query(
    UPDATE stripe_events SET processed = true, processed_at = now() WHERE stripe_event_id = $1,
    [event.id],
  );
}

Highlights of the Implementation
1. Strong typing and configuration
Environment loader (src/lib/config.ts): Loads environment variables, validates required keys, and exports a typed configuration object.

Database client (src/lib/db.ts): Creates a single connection pool using pg.Pool and exports a generic query<T> helper for typed queries.

Typed tables (src/types/db.ts): Defines TypeScript interfaces for Users, Jobs, JobEvents, CreditLedger, PaymentIntents, Payouts, StripeEvents and Disputes, alongside the JobStatus and UserRole unions. Custom declarations (src/types/pg.d.ts): Provides minimal type definitions for the pg module to satisfy the TypeScript compiler without installing @types/pg.

2. State machine
A pure state machine (src/lib/stateMachine.ts) encapsulates allowed transitions for a job. It defines

actions: accept, on_my_way, check_in, check_out, submit_for_approval, approve, dispute, cancel, and a placeholder create;

allowed actors for each action (client, cleaner, admin, system);

transition rules mapping current statuses to next statuses;

optional extraChecks such as requiring GPS on check‑in, ratings for approval, or notes for disputes.

Attempting an invalid transition results in a descriptive error.

3. Event publishing
A centralized event helper (src/lib/events.ts) persists job events to the job_events table via the shared query helper and logs them. The routes and services call publishEvent every time a job is created or its status changes.

4. Jobs service
The jobs service (src/services/jobs.ts) handles business logic around jobs:

createJob inserts a job with status requested and publishes a job.created event.

performAction loads the job, applies the state machine to compute the next status, updates the job, and publishes the corresponding event (job.accepted, job.on_my_way, etc.). It also records actual_start_at and actual_end_at during check‑in/out.

Functions exist to list jobs for clients, cleaners and admins, and to fetch jobs by id.

POST /jobs: create a job (client only).

GET /jobs: list jobs (client sees their jobs, cleaner sees assigned jobs, admin sees all).

GET /jobs/:id: fetch a job (only if client, assigned cleaner or admin).

POST /jobs/:id/accept, /on_my_way, /check_in, /check_out, /submit_for_approval, /approve, /dispute, /cancel: transition endpoints using the state machine, with proper role checks and request validation.

src/routes/stripe.ts — stub webhook endpoint that logs incoming events. The file is ready to integrate with a future payments service.

src/routes/health.ts — unchanged; returns application health and uptime.

import { Pool } from '@neondatabase/serverless';
import { config } from './config';

export const pool = new Pool({ connectionString: config.databaseUrl });

export async function query<T = unknown>(text: string, params: unknown[] = []): Promise<T[]> {
  const client = await pool.connect();
  try {
    const result = await client.query(text, params);
    return result.rows as T[];
  } finally {
    client.release();
  }
}

import { Request, Response, NextFunction } from 'express';
import jwt from 'jsonwebtoken';
import { config } from './config';

export type UserRole = 'client' | 'cleaner' | 'admin';

export interface AuthUser {
  id: string;
  role: UserRole;
}

declare global {
  namespace Express {
    interface Request {
      user?: AuthUser;
    }
  }
}

// Attach user if a valid JWT is present, but don't enforce auth
export function authMiddlewareAttachUser(req: Request, _res: Response, next: NextFunction) {
  const header = req.headers.authorization;
  if (!header?.startsWith('Bearer ')) {
    return next();
  }
  try {
    const token = header.slice('Bearer '.length);
    const decoded = jwt.verify(token, config.jwtSecret) as AuthUser;
    req.user = decoded;
  } catch {
    // ignore invalid
  }
  next();
}

// Enforce auth, optionally by role
export function auth(requiredRole?: UserRole) {
  return (req: Request, res: Response, next: NextFunction) => {
    const header = req.headers.authorization;
    if (!header?.startsWith('Bearer ')) {
      return res.status(401).json({ error: 'Missing auth token' });
    }
    try {
      const token = header.slice('Bearer '.length);
      const decoded = jwt.verify(token, config.jwtSecret) as AuthUser;
      req.user = decoded;

      if (requiredRole && decoded.role !== requiredRole) {
        return res.status(403).json({ error: 'Forbidden' });
      }

      next();
    } catch {
      return res.status(401).json({ error: 'Invalid auth token' });
    }
  };
}

import { ZodSchema } from 'zod';
import { Request, Response, NextFunction } from 'express';

export function validateBody<T>(schema: ZodSchema<T>) {
  return (req: Request, res: Response, next: NextFunction) => {
    const result = schema.safeParse(req.body);
    if (!result.success) {
      return res.status(400).json({
        error: 'Invalid body',
        details: result.error.issues,
      });
    }
    req.body = result.data;
    next();
  };
}

import type { JobStatus } from '../types/db';

export type JobAction =
  | 'create'
  | 'accept'
  | 'on_my_way'
  | 'check_in'
  | 'check_out'
  | 'submit_for_approval'
  | 'approve'
  | 'dispute'
  | 'cancel';

export type Actor = 'client' | 'cleaner' | 'admin' | 'system';

export interface TransitionContext {
  actor: Actor;
  hasGps?: boolean;
  rating?: number;
  hasClientNotes?: boolean;
}

interface TransitionRule {
  from: JobStatus[];
  to: JobStatus;
  allowedActors: Actor[];
  extraChecks?: (ctx: TransitionContext) => void;
}

const rules: Record<JobAction, TransitionRule> = {
  accept: {
    from: ['requested'],
    to: 'accepted',
    allowedActors: ['cleaner'],
  },
  on_my_way: {
    from: ['accepted'],
    to: 'on_my_way',
    allowedActors: ['cleaner'],
  },
  check_in: {
    from: ['on_my_way'],
    to: 'in_progress',
    allowedActors: ['cleaner'],
    extraChecks: (ctx) => {
      if (!ctx.hasGps) throw new Error('GPS required to check_in');
    },
  },
  check_out: {
    from: ['in_progress'],
    to: 'awaiting_approval',
    allowedActors: ['cleaner'],
  },
  submit_for_approval: {
    from: ['awaiting_approval'],
    to: 'awaiting_approval',
    allowedActors: ['cleaner'],
  },
  approve: {
    from: ['awaiting_approval', 'disputed'],
    to: 'completed',
    allowedActors: ['client', 'admin'],
    extraChecks: (ctx) => {
      if (ctx.rating && (ctx.rating < 1 || ctx.rating > 5)) {
        throw new Error('Rating must be between 1 and 5');
      }
    },
  },
  dispute: {
    from: ['awaiting_approval'],
    to: 'disputed',
    allowedActors: ['client', 'admin'],
    extraChecks: (ctx) => {
      if (!ctx.hasClientNotes) {
        throw new Error('Client notes required to dispute');
      }
    },
  },
  cancel: {
    from: ['requested', 'accepted', 'on_my_way'],
    to: 'cancelled',
    allowedActors: ['client', 'admin'],
  },
  create: {
    from: ['requested'],
    to: 'requested',
    allowedActors: ['client', 'system'],
  },
};

export function transitionJob(
  current: JobStatus,
  action: JobAction,
  ctx: TransitionContext,
): JobStatus {
  const rule = rules[action];
  if (!rule) throw new Error(Unsupported action: ${action});

  if (!rule.from.includes(current)) {
    throw new Error(Cannot ${action} when status is ${current});
  }

  if (!rule.allowedActors.includes(ctx.actor)) {
    throw new Error(Actor ${ctx.actor} not allowed to ${action});
  }

  rule.extraChecks?.(ctx);

  return rule.to;
}

import { query } from './db';
import { logger } from './logger';
import { config } from './config';

export type JobEventType =
  | 'job.created'
  | 'job.accepted'
  | 'job.on_my_way'
  | 'job.checked_in'
  | 'job.checked_out'
  | 'job.submitted_for_approval'
  | 'job.approved'
  | 'job.disputed'
  | 'job.cancelled'
  | 'job.auto_cancelled'
  | 'job.overridden'
  | 'job.payment_succeeded';

export interface PublishEventInput {
  jobId: string;
  actorType: 'client' | 'cleaner' | 'admin' | 'system';
  actorId: string | null;
  eventType: JobEventType;
  payload?: Record<string, any>;
}

export async function publishEvent(input: PublishEventInput): Promise<void> {
  const payloadJson = JSON.stringify(input.payload ?? {});

  await query(
    
      INSERT INTO job_events (job_id, actor_type, actor_id, event_type, payload)
      VALUES ($1, $2, $3, $4, $5)
    ,
    [input.jobId, input.actorType, input.actorId, input.eventType, payloadJson],
  );

  logger.info('job_event', {
    jobId: input.jobId,
    eventType: input.eventType,
    actorType: input.actorType,
    actorId: input.actorId,
  });

  // Optional: forward to n8n
  if (config.n8nWebhookUrl) {
    postJson(config.n8nWebhookUrl, {
      jobId: input.jobId,
      eventType: input.eventType,
      actorType: input.actorType,
      actorId: input.actorId,
      payload: input.payload ?? {},
    }).catch((err) => {
      logger.error('n8n_forward_failed', { error: (err as Error).message });
    });
  }
}

src/lib/httpClient.ts
import http from 'http';
import https from 'https';
import { URL } from 'url';

export function postJson(url: string, body: any): Promise<void> {
  return new Promise((resolve, reject) => {
    const parsed = new URL(url);
    const data = Buffer.from(JSON.stringify(body), 'utf-8');

    const isHttps = parsed.protocol === 'https:';
    const client = isHttps ? https : http;

    const req = client.request(
      {
        hostname: parsed.hostname,
        port: parsed.port || (isHttps ? 443 : 80),
        path: parsed.pathname + parsed.search,
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
          'Content-Length': data.length,
        },
      },
      (res) => {
        // consume and ignore body
        res.on('data', () => {});
        res.on('end', () => resolve());
      },
    );

    req.on('error', reject);
    req.write(data);
    req.end();
  });
}

interface ListJobsFilters {
  status?: string;
  cleanerId?: string;
  clientId?: string;
  from?: string;
  to?: string;
}

async function listJobs(filters: ListJobsFilters): Promise<Job[]> {
  const conditions: string[] = [];
  const params: any[] = [];
  let idx = 1;

  if (filters.status) {
    conditions.push(status = $${idx++});
    params.push(filters.status);
  }
  if (filters.cleanerId) {
    conditions.push(cleaner_id = $${idx++});
    params.push(filters.cleanerId);
  }
  if (filters.clientId) {
    conditions.push(client_id = $${idx++});
    params.push(filters.clientId);
  }
  if (filters.from) {
    conditions.push(created_at >= $${idx++});
    params.push(filters.from);
  }
  if (filters.to) {
    conditions.push(created_at <= $${idx++});
    params.push(filters.to);
  }

  const where = conditions.length ? WHERE ${conditions.join(' AND ')} : '';

  return query<Job>(
    
      SELECT *
      FROM jobs
      ${where}
      ORDER BY created_at DESC
    ,
    params,
  );
}

import { config } from '../lib/config';

export async function createPaymentForJob(
  jobId: string,
  amountCents: number,
  clientStripeCustomerId: string,
): Promise<Stripe.PaymentIntent> {
  const paymentIntent = await stripe.paymentIntents.create({
    amount: amountCents,
    currency: 'usd',
    customer: clientStripeCustomerId,
    metadata: { jobId },
  });

  await query(
    
      INSERT INTO payment_intents (job_id, stripe_payment_intent_id, status, amount_cents, currency, created_at, updated_at)
      VALUES ($1, $2, $3, $4, $5, now(), now())
    ,
    [jobId, paymentIntent.id, paymentIntent.status, amountCents, 'usd'],
  );

  return paymentIntent;
}

export async function handleStripeEvent(event: Stripe.Event): Promise<void> {
  await query(
    
      INSERT INTO stripe_events (stripe_event_id, type, payload, processed, created_at)
      VALUES ($1, $2, $3, false, now())
      ON CONFLICT (stripe_event_id) DO NOTHING
    ,
    [event.id, event.type, JSON.stringify(event)],
  );

  switch (event.type) {
    case 'payment_intent.succeeded': {
      const pi = event.data.object as Stripe.PaymentIntent;
      const jobId = pi.metadata.jobId;
      if (jobId) {
        await query<DbPaymentIntent>(
          
            UPDATE payment_intents
            SET status = $2, updated_at = now()
            WHERE stripe_payment_intent_id = $1
            RETURNING *
          ,
          [pi.id, pi.status],
        );
        await publishEvent({
          jobId,
          actorType: 'system',
          actorId: null,
          eventType: 'job.payment_succeeded',
          payload: { stripePaymentIntentId: pi.id },
        });
      }
      break;
    }
    default:
      break;
  }

Short answer: No, you don’t have to.

Backend = our PureTask Node/TypeScript API

Frontend = one Base44 project that talks to that backend

Client web app (booking + account)

Admin dashboard (overrides, disputes, metrics)

✅ PureTask backend (Node + Neon) + Base44 as the only app UI

No separate Next.js app is required to “make it work.”

3️⃣ How they connect in practice (concrete example)
Let’s say a client books a cleaning in Base44:

Client fills a form in Base44:

Date/time, address, size, notes, etc.

The Base44 page calls your backend:

POST https://api.puretask.com/jobs

Headers: Authorization: Bearer <client JWT>

Your backend:

Validates data (Zod)

Creates a row in jobs in Neon

Inserts job_event job.created

Maybe creates a Stripe PaymentIntent

Base44:

Shows “Booking confirmed” UI

All that logic is in the backend; Base44 just calls the endpoints.
