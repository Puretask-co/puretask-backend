2. PaymentIntents endpoint
We’ll create a route file stripeRoutes.js that exports an Express router.

2.1. stripeRoutes.js skeleton
Create stripeRoutes.js in /server:

const express = require('express');
const bodyParser = require('body-parser');

module.exports = (stripe) => {
  const router = express.Router();

  // 1) Create PaymentIntent
  router.post('/payments/create-intent', async (req, res) => {
    try {
      const { amount, currency = 'usd', bookingId, clientId } = req.body;

      if (!amount || !bookingId || !clientId) {
        return res.status(400).json({ error: 'Missing required fields' });
      }

      // amount is in cents (e.g., $10.00 => 1000)
      const paymentIntent = await stripe.paymentIntents.create({
        amount,
        currency,
        metadata: {
          bookingId,
          clientId,
        },
      });

      res.json({
        clientSecret: paymentIntent.client_secret,
        paymentIntentId: paymentIntent.id,
      });
    } catch (err) {
      console.error('Error creating PaymentIntent:', err);
      res.status(500).json({ error: 'Internal server error' });
    }
  });

  return router;
};
This endpoint:

Accepts amount in cents (so your React app converts dollars/credits → cents).

Includes metadata bookingId, clientId for your Base44 entities.

Returns clientSecret for the frontend to confirm the payment.
