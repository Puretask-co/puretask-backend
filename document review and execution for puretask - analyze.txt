-- PureTask • Section 2: Job Lifecycle Orchestrator
-- Creates enums and helper functions used by the Job Orchestrator.

DO $$ BEGIN
  CREATE TYPE job_status AS ENUM (
    'created','accepted','in_progress','completed',
    'awaiting_client','approved','disputed','cancelled'
  );
EXCEPTION WHEN duplicate_object THEN NULL; END $$;

DO $$ BEGIN
  CREATE TYPE cleaning_type AS ENUM ('basic','deep','moveout');
EXCEPTION WHEN duplicate_object THEN NULL; END $$;

DO $$ BEGIN
  CREATE TYPE job_event_type AS ENUM (
    'job.created','job.accepted','job.checkin','job.checkout','job.completed',
    'job.awaiting_client','job.approved','job.disputed','job.cancelled',
    'job.topup_required','job.topup_confirmed','job.auto_approved'
  );
EXCEPTION WHEN duplicate_object THEN NULL; END $$;

-- 15-minute rounding helper. When round_up=true, always round up to next 0.25 hour.
CREATE OR REPLACE FUNCTION round_hours_15min(seconds BIGINT, round_up BOOLEAN DEFAULT false)
RETURNS NUMERIC AS $$
DECLARE hrs NUMERIC;
BEGIN
  hrs := seconds / 3600.0;
  IF round_up THEN
    RETURN CEIL(hrs * 4)::NUMERIC / 4.0;
  ELSE
    RETURN ROUND(hrs * 4) / 4.0;
  END IF;
END; $$ LANGUAGE plpgsql IMMUTABLE;
""").strip()

-- 005_jobs_compute_functions.sql
-- Calculation helpers used by the workflow.

-- Compute final charge in credits given cleaning type, hours, and snapshot rates
CREATE OR REPLACE FUNCTION calc_job_charge_credits(
  cleaning cleaning_type,
  hours NUMERIC,
  base_cph NUMERIC,
  deep_cph NUMERIC,
  moveout_cph NUMERIC
) RETURNS NUMERIC AS $$
DECLARE addon NUMERIC := 0;
BEGIN
  IF cleaning = 'deep' THEN
    addon := COALESCE(deep_cph, 0);
  ELSIF cleaning = 'moveout' THEN
    addon := COALESCE(moveout_cph, 0);
  END IF;
  RETURN (COALESCE(base_cph,0) + addon) * COALESCE(hours,0);
END; $$ LANGUAGE plpgsql IMMUTABLE;

-- Recompute job.total_seconds_worked from logs and persist
CREATE OR REPLACE FUNCTION recompute_job_seconds(job_id_in TEXT) RETURNS BIGINT AS $$
DECLARE total BIGINT;
BEGIN
  SELECT COALESCE(SUM(segment_seconds),0) INTO total FROM job_time_logs WHERE job_id = job_id_in;
  UPDATE jobs SET total_seconds_worked = total WHERE id = job_id_in;
  RETURN total;
END; $$ LANGUAGE plpgsql;
""").strip()
