-- 15-min rounding helper (rounds UP to next 0.25 hr when flag=true)
CREATE OR REPLACE FUNCTION round_hours_15min(
  seconds BIGINT,
  round_up BOOLEAN DEFAULT false
) RETURNS NUMERIC AS $$
DECLARE hrs NUMERIC;
BEGIN
  hrs := seconds / 3600.0;
  IF round_up THEN
    RETURN CEIL(hrs * 4)::NUMERIC / 4.0;
  ELSE
    RETURN ROUND(hrs * 4) / 4.0;
  END IF;
END; $$ LANGUAGE plpgsql IMMUTABLE;

005_jobs_compute_functions.sql
-- Compute charge in credits based on snapshots & actual hours
-- cleaning_type determines which addon rate applies (deep/moveout), else 0
CREATE OR REPLACE FUNCTION calc_job_charge_credits(
  cleaning cleaning_type,
  hours NUMERIC,
  base_cph NUMERIC,
  deep_cph NUMERIC,
  moveout_cph NUMERIC
) RETURNS NUMERIC AS $$
DECLARE addon NUMERIC := 0;
BEGIN
  IF cleaning = 'deep' THEN
    addon := COALESCE(deep_cph,0);
  ELSIF cleaning = 'moveout' THEN
    addon := COALESCE(moveout_cph,0);
  END IF;
  RETURN (COALESCE(base_cph,0) + addon) * COALESCE(hours,0);
END; $$ LANGUAGE plpgsql IMMUTABLE;

-- Convenience: recompute total_seconds_worked from job_time_logs
CREATE OR REPLACE FUNCTION recompute_job_seconds(job_id_in TEXT)
RETURNS BIGINT AS $$
DECLARE total BIGINT;
BEGIN
  SELECT COALESCE(SUM(segment_seconds),0)
    INTO total
  FROM job_time_logs
  WHERE job_id = job_id_in;

  UPDATE jobs
     SET total_seconds_worked = total
   WHERE id = job_id_in;

  RETURN total;
END; $$ LANGUAGE plpgsql;

4) Business rules, formulas & edge cases

Time rounding: 15-minute increments (round_hours_15min(seconds, true) rounds up). You can set require_round_up=false to round to nearest.

Final charge formula (credits):
final = (base_cph + addon_cph) * hours_actual

addon_cph = deep_cph if cleaning_type=deep, or moveout_cph if moveout, else 0.

Refund credits (computed in Section-3): refund = held_credits − final_charge_credits (≥ 0).

Top-up (insufficient hold mid-job): your In-Progress screen should watch elapsed value vs held_credits; if low, call Section-3 top-up hold. (We’ll implement that in Section-3. This workflow just logs and continues.)

GPS tolerance: we log lat/lng; add distance checks in app level (optionally compute distance_m in job_gps_logs).

Photos: upload to your storage/CDN and call a tiny endpoint (or a direct DB write) to add job_photos rows (before/after). This workflow doesn’t enforce them; it records presence via Disputes/Notifications later.

Auto-approve: Wait node is 24h (86,400,000 ms). Change if needed.

-- 15-min rounding helper (rounds UP to next 0.25 hr when flag=true)
CREATE OR REPLACE FUNCTION round_hours_15min(
  seconds BIGINT,
  round_up BOOLEAN DEFAULT false
) RETURNS NUMERIC AS $$
DECLARE hrs NUMERIC;
BEGIN
  hrs := seconds / 3600.0;
  IF round_up THEN
    RETURN CEIL(hrs * 4)::NUMERIC / 4.0;
  ELSE
    RETURN ROUND(hrs * 4) / 4.0;
  END IF;
END; $$ LANGUAGE plpgsql IMMUTABLE;

-- 005_jobs_compute_functions.sql

-- Compute charge in credits based on snapshots & actual hours
-- cleaning_type determines which addon rate applies (deep/moveout), else 0
CREATE OR REPLACE FUNCTION calc_job_charge_credits(
  cleaning cleaning_type,
  hours NUMERIC,
  base_cph NUMERIC,
  deep_cph NUMERIC,
  moveout_cph NUMERIC
) RETURNS NUMERIC AS $$
DECLARE addon NUMERIC := 0;
BEGIN
  IF cleaning = 'deep' THEN
    addon := COALESCE(deep_cph,0);
  ELSIF cleaning = 'moveout' THEN
    addon := COALESCE(moveout_cph,0);
  END IF;
  RETURN (COALESCE(base_cph,0) + addon) * COALESCE(hours,0);
END; $$ LANGUAGE plpgsql IMMUTABLE;

-- Convenience: recompute total_seconds_worked from job_time_logs
CREATE OR REPLACE FUNCTION recompute_job_seconds(job_id_in TEXT)
RETURNS BIGINT AS $$
DECLARE total BIGINT;
BEGIN
  SELECT COALESCE(SUM(segment_seconds),0)
    INTO total
  FROM job_time_logs
  WHERE job_id = job_id_in;

  UPDATE jobs
     SET total_seconds_worked = total
   WHERE id = job_id_in;

  RETURN total;
END; $$ LANGUAGE plpgsql;
